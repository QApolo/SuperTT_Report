\paragraph{Trayectoria Principal}
\label{CUA_Crear_Proyecto}
	\begin{enumerate}
	    \item {[Actor]} Presiona el botón \textit{Crear Proyecto} en la pantalla \textbf{\nameref{fig:MIUA-5}}.
	    
	    \item {[Sistema]} Muestra un nuevo recuadro en la pantalla \textbf{\nameref{fig:MIUA-5}} solicitando el nombre del nuevo proyecto.
	    
	    \item {[Actor]} Ingresa el nombre del nuevo proyecto.
	    
	    \item {[Actor]} Solicita guardar el nuevo proyecto oprimiendo el botón \textit{Aceptar} de la pantalla \textbf{\nameref{fig:MIUA-5}}. \hyperref[A-PR-CU1.2:TA]{[Trayectoria A]}
	    
	    \item {[Sistema]} Verifica que el nombre del proyecto sea válido como se especifica en las reglas de negocio REGLAS-DE-NEGOCIO. \hyperref[A-PR-CU1.2:TB]{[Trayectoria B]}
	    
	    \item {[Sistema]} Registra el nuevo proyecto en el sistema.
	    
	    \item {[Sistema]} Muestra la pantalla \textbf{\nameref{fig:MIUA-5}} y actualiza la lista de proyectos.

	\end{enumerate}
	Fin del caso de uso.
	
\paragraph{Trayectoria Alternativa A} \label{A-PR-CU1.2:TA}
	El actor cancela la operación.
	%para más trayectorias, copiar y pegar toda esta sección y en el comando siguiente cambiar por la letra correspondiente
	\begin{enumerate}[label=A\arabic*.]
		\item {[Actor]} Solicita cancelar la operación oprimiendo el botón \textit{Cancelar} en la pantalla emergente de \textbf{\nameref{fig:MIUA-5}}.
		\item {[Sistema]} Muestra la pantalla \textbf{\nameref{fig:MIUA-5}}.
	\end{enumerate}
	Fin de la trayectoria alternativa.
